#include <bits/stdc++.h>
using namespace std;

void coin(int n, int c, int w[])
{
   int dp[n + 1][c + 1];
   int i, j;
   for (i = 0; i <= n; i++)
   {
      for (j = 0; j <= c; j++)
      {
         if (i == 0 || j == 0)
            dp[i][j] = 1;
         else
         {
            if (w[i] > j)
               dp[i][j] = dp[i - 1][j];
            else
               dp[i][j] = dp[i - 1][j] + dp[i][j - w[i]];
         }
         printf("%d ", dp[i][j]);
      }
      printf("\n");
   }
   printf("Maximum ways = %d\n", dp[n][c]);
}

int main()
{
   int n, i, c;
   printf("Enter No Of Item: ");
   scanf("%d", &n);
   int w[n];
   for (i = 0; i < n; i++)
      scanf("%d", &w[i]);
   printf("Enter maximum Capacity: ");
   cin >> c;
   coin(n, c, w);

   return 0;
}
